CODE

    abstract class Employee {
        abstract double calculateSalary();
        abstract void displayInfo();
    }
    
    class Manager extends Employee {
        @Override
        double calculateSalary() {
            return 70000;
        }
    
        @Override
        void displayInfo() {
            System.out.println("Manager with salary: " + calculateSalary());
        }
    }
    
    class Programmer extends Employee {
        @Override
        double calculateSalary() {
            return 50000;
        }
    
        @Override
        void displayInfo() {
            System.out.println("Programmer with salary: " + calculateSalary());
        }
    }
    
    public class Main {
        public static void main(String[] args) {
            Employee manager = new Manager();
            manager.displayInfo();
    
            Employee programmer = new Programmer();
            programmer.displayInfo();
        }
    }
    
    
OUTPUT

    Manager with salary: 70000.0
    Programmer with salary: 50000.0
