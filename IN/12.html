    class Parent {
        Parent() {
            System.out.println("Parent class constructor called");
        }
    }
    
    class Child extends Parent {
        Child() {
            super();
            System.out.println("Child class constructor called");
        }
    }
    
    public class Main {
        public static void main(String[] args) {
            new Child();
        }
    }
