CODE

    class Animal {
        String name = "Animal";
    }
    
    class Dog extends Animal {
        String name = "Dog";
    
        public void printName() {
            System.out.println("Superclass name: " + super.name);
            System.out.println("Subclass name: " + name);
        }
    }
    
    public class Main {
        public static void main(String[] args) {
            Dog dog = new Dog();
            dog.printName();
        }
    }
    
    
OUTPUT

    Superclass name: Animal
    Subclass name: Dog
