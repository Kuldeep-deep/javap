CODE

    class Animal {
        public void makeSound() {
            System.out.println("Animal makes sound");
        }
    }
    
    class Dog extends Animal {
        @Override
        public void makeSound() {
            System.out.println("Dog barks");
        }
    }
    
    public class Main {
        public static void main(String[] args) {
            Animal animal = new Dog();
            animal.makeSound();
        }
    }
    
    
OUTPUT

    Dog barks
