CODE

    class Shape {
        public void draw() {
            System.out.println("Drawing a shape");
        }
    }
    
    class Circle extends Shape {
        @Override
        public void draw() {
            System.out.println("Drawing a circle");
        }
    }
    
    class Square extends Shape {
        @Override
        public void draw() {
            System.out.println("Drawing a square");
        }
    }
    
    public class Main {
        public static void main(String[] args) {
            Shape shape;
    
            shape = new Circle();
            shape.draw();
    
            shape = new Square();
            shape.draw();
        }
    }
    
    
OUTPUT

    Drawing a circle
    Drawing a square
