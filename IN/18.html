CODE

    abstract class Animal {
        abstract void eat();
        abstract void sleep();
    }
    
    class Lion extends Animal {
        @Override
        void eat() {
            System.out.println("Lion eats meat.");
        }
    
        @Override
        void sleep() {
            System.out.println("Lion sleeps 10 hours a day.");
        }
    }
    
    class Deer extends Animal {
        @Override
        void eat() {
            System.out.println("Deer eats grass.");
        }
    
        @Override
        void sleep() {
            System.out.println("Deer sleeps 6 hours a day.");
        }
    }
    
    public class Main {
        public static void main(String[] args) {
            Animal lion = new Lion();
            lion.eat();
            lion.sleep();
    
            Animal deer = new Deer();
            deer.eat();
            deer.sleep();
        }
    }
    
    
OUTPUT

    Lion eats meat.
    Lion sleeps 10 hours a day.
    Deer eats grass.
    Deer sleeps 6 hours a day.
