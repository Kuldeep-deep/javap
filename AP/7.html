    import java.applet.Applet;
    import java.awt.Graphics;
    
    public class DrawConcentricCirclesApplet extends Applet {
        public void paint(Graphics g) {
            for (int i = 1; i <= 4; i++) {
                g.drawOval(50 - i * 10, 50 - i * 10, 100 + i * 20, 100 + i * 20);
            }
        }
    }
