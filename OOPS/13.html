    import java.util.HashMap;

    public class Inventory {
        HashMap products = new HashMap<>();
    
        public void addProduct(String product, int quantity) {
            products.put(product, quantity);
        }
    
        public void removeProduct(String product) {
            products.remove(product);
        }
    
        public void checkLowInventory() {
            products.forEach((product, quantity) -> {
                if (quantity < 5) {
                    System.out.println("Low inventory for: " + product);
                }
            });
        }
    
        public void displayInventory() {
            products.forEach((product, quantity) -> {
                System.out.println("Product: " + product + ", Quantity: " + quantity);
            });
        }
    
        public static void main(String[] args) {
            Inventory inventory = new Inventory();
            inventory.addProduct("Laptop", 3);
            inventory.addProduct("Mouse", 10);
            inventory.addProduct("Keyboard", 2);
    
            inventory.displayInventory();
            inventory.checkLowInventory();
        }
    }
    
    
OUTPUT

    Product: Laptop, Quantity: 3
    Product: Mouse, Quantity: 10
    Product: Keyboard, Quantity: 2
    Low inventory for: Laptop
    Low inventory for: Keyboard
