    import java.time.LocalDate;
    import java.time.temporal.ChronoUnit;
    
    public class Employee {
        String name;
        double salary;
        LocalDate hireDate;
    
        public Employee(String name, double salary, LocalDate hireDate) {
            this.name = name;
            this.salary = salary;
            this.hireDate = hireDate;
        }
    
        public long yearsOfService() {
            return ChronoUnit.YEARS.between(hireDate, LocalDate.now());
        }
    
        public void display() {
            System.out.println("Name: " + name + ", Salary: $" + salary + ", Years of Service: " + yearsOfService());
        }
    
        public static void main(String[] args) {
            Employee emp = new Employee("John", 75000, LocalDate.of(2015, 5, 15));
            emp.display();
        }
    }
    
    
OUTPUT

    Name: John, Salary: $75000.0, Years of Service: 8
