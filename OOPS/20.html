    class Sample {
        String message;
    
        public Sample() {
            message = "Hello from default constructor!";
        }
    
        public void displayMessage() {
            System.out.println(message);
        }
    
        public static void main(String[] args) {
            Sample obj = new Sample();
            obj.displayMessage();
        }
    }
    
    
OUTPUT

    Hello from default constructor!
