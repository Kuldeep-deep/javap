    class Student {
        String name;
        int age;
    
        public Student(String name, int age) {
            this.name = name;
            this.age = age;
        }
    
        public void displayInfo() {
            System.out.println("Name: " + name + ", Age: " + age);
        }
    
        public static void main(String[] args) {
            Student student = new Student("Alice", 20);
            student.displayInfo();
        }
    }
    
    
OUTPUT

    Name: Alice, Age: 20
