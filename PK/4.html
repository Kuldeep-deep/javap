CODE

    import java.util.Date;

    class MyDate {
        private Date date;
    
        public MyDate() {
            date = new Date();
        }
    
        public void printDate() {
            System.out.println("Current date and time: " + date);
        }
    }
    
    public class Main {
        public static void main(String[] args) {
            MyDate myDate = new MyDate();
            myDate.printDate();
        }
    }
    
    
OUTPUT

    Current date and time: Thu Nov 08 15:25:57 IST 2024
