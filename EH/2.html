    public class OddNumberExceptionExample {

        public static void main(String[] args) {
            try {
                checkEvenNumber(5);
            } catch (Exception e) {
                System.out.println("Exception caught: " + e);
            }
        }
    
        public static void checkEvenNumber(int number) throws Exception {
            if (number % 2 != 0) {
                throw new Exception("The number " + number + " is odd, and an exception was thrown.");
            } else {
                System.out.println("The number " + number + " is even.");
            }
        }
    }
    
    
OUTPUT

    Exception caught: The number 5 is odd, and an exception was thrown.
