    import java.io.File;
    import java.io.FileNotFoundException;
    import java.util.Scanner;
    
    public class EmptyFileExceptionExample {
    
        public static void checkIfFileIsEmpty(String fileName) throws Exception {
            File file = new File(fileName);
            
            if (file.length() == 0) {
                throw new Exception("The file is empty");
            }
    
            Scanner scanner = new Scanner(file);
            while (scanner.hasNextLine()) {
                System.out.println(scanner.nextLine());
            }
            scanner.close();
        }
    
        public static void main(String[] args) {
            String fileName = "sample.txt";
    
            try {
                checkIfFileIsEmpty(fileName);
                System.out.println("File content is read successfully.");
            } catch (Exception e) {
                System.out.println("Exception caught: " + e.getMessage());
            }
        }
    }
    
    
OUTPUT

    Exception caught: The file is empty
