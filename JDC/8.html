    public static void scrollableUpdatableResultSet(Connection conn) throws SQLException {
        String selectSQL = "SELECT * FROM Employee";
        try (Statement stmt = conn.createStatement(ResultSet.TYPE_SCROLL_INSENSITIVE, ResultSet.CONCUR_UPDATABLE);
             ResultSet rs = stmt.executeQuery(selectSQL)) {
            
            
            rs.moveToInsertRow();
            rs.updateString("name", "Jane Doe");
            rs.updateInt("age", 28);
            rs.updateDouble("salary", 60000);
            rs.insertRow();
            System.out.println("Row inserted.");
            
            
            rs.absolute(1);
            rs.updateDouble("salary", 65000);
            rs.updateRow();
            System.out.println("Row updated.");
            
            
            rs.absolute(2);
            rs.deleteRow();
            System.out.println("Row deleted.");
        }
    }
