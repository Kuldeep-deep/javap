    public static void preparedStatements(Connection conn) throws SQLException {
        
        String insertSQL = "INSERT INTO Employee (name, age, salary) VALUES (?, ?, ?)";
        try (PreparedStatement pstmt = conn.prepareStatement(insertSQL)) {
            pstmt.setString(1, "Alice");
            pstmt.setInt(2, 35);
            pstmt.setDouble(3, 70000);
            pstmt.executeUpdate();
            System.out.println("Row inserted using PreparedStatement.");
        }
    
        
        String updateSQL = "UPDATE Employee SET salary = ? WHERE name = ?";
        try (PreparedStatement pstmt = conn.prepareStatement(updateSQL)) {
            pstmt.setDouble(1, 75000);
            pstmt.setString(2, "Alice");
            pstmt.executeUpdate();
            System.out.println("Row updated using PreparedStatement.");
        }
    
        
        String deleteSQL = "DELETE FROM Employee WHERE name = ?";
        try (PreparedStatement pstmt = conn.prepareStatement(deleteSQL)) {
            pstmt.setString(1, "Alice");
            pstmt.executeUpdate();
            System.out.println("Row deleted using PreparedStatement.");
        }
    }
