CODE

    import java.util.Arrays;

    public class SortArrayAscending {
        public static void main(String[] args) {
            int[] arr = {5, 3, 8, 1, 2};
    
            Arrays.sort(arr);
    
            System.out.println("Array sorted in ascending order:");
            for (int num : arr) {
                System.out.print(num + " ");
            }
        }
    }
    
    
OUTPUT

    Array sorted in ascending order:
    1 2 3 5 8
