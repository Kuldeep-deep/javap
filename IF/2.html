CODE

    interface Animal {
        void bark();
    }
    
    class Dog implements Animal {
        public void bark() {
            System.out.println("Dog is barking");
        }
    }
    
    public class Main {
        public static void main(String[] args) {
            Animal myDog = new Dog();
            myDog.bark();
        }
    }
    
    
OUTPUT

    Dog is barking
