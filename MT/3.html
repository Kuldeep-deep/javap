    import java.util.Arrays;

    public class MultiThreadSort {
        public static void main(String[] args) throws InterruptedException {
            int[] array = {7, 3, 1, 9, 5, 4, 8, 6, 2, 0};
            int mid = array.length / 2;
    
            Thread sortFirstHalf = new Thread(() -> Arrays.sort(array, 0, mid));
            Thread sortSecondHalf = new Thread(() -> Arrays.sort(array, mid, array.length));
    
            sortFirstHalf.start();
            sortSecondHalf.start();
    
            sortFirstHalf.join();
            sortSecondHalf.join();
    
            merge(array, mid);
            System.out.println("Sorted array: " + Arrays.toString(array));
        }
    
        public static void merge(int[] array, int mid) {
            int[] temp = new int[array.length];
            int i = 0, j = mid, k = 0;
    
            while (i < mid && j < array.length) {
                if (array[i] < array[j]) temp[k++] = array[i++];
                else temp[k++] = array[j++];
            }
    
            while (i < mid) temp[k++] = array[i++];
            while (j < array.length) temp[k++] = array[j++];
    
            System.arraycopy(temp, 0, array, 0, array.length);
        }
    }
