    public class CountTwos {

        public static void main(String[] args) {
            int number = 225632;
            System.out.println("Number of 2's in " + number + ": " + countTwos(number));
        }
    
        public static int countTwos(int number) {
            int count = 0;
            while (number > 0) {
                int digit = number % 10;
                if (digit == 2) {
                    count++;
                }
                number /= 10;
            }
            return count;
        }
    }
