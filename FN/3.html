    import java.util.Scanner;

    public class MiddleCharacter {
    
        public static void main(String[] args) {
            Scanner scanner = new Scanner(System.in);
    
            System.out.print("Enter a string: ");
            String input = scanner.nextLine();
    
            String middleChars = getMiddleCharacter(input);
            System.out.println("Middle character(s): " + middleChars);
    
            scanner.close();
        }
    
        public static String getMiddleCharacter(String str) {
            int length = str.length();
            int middleIndex = length / 2;
    
            if (length % 2 == 0) {
                return String.valueOf(str.charAt(middleIndex - 1));
            } else {
                return str.charAt(middleIndex - 1) + "" + str.charAt(middleIndex);
            }
        }
    }
