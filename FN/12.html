    public class MiddlePointChecker {

        public static void main(String[] args) {
            System.out.println(isMiddlePoint(5, 10, 15)); // true
            System.out.println(isMiddlePoint(20, 10, 30)); // true
            System.out.println(isMiddlePoint(10, 20, 30)); // false
        }
    
        public static boolean isMiddlePoint(int a, int b, int c) {
            return (a > b && a < c) || (a < b && a > c) || 
                   (b > a && b < c) || (b < a && b > c) || 
                   (c > a && c < b) || (c < a && c > b);
        }
    }
