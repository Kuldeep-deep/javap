    import java.util.Arrays;

    public class ConsecutiveChecker {
    
        public static void main(String[] args) {
            System.out.println(areConsecutive(3, 5, 4)); 
            System.out.println(areConsecutive(10, 8, 9)); 
            System.out.println(areConsecutive(1, 3, 5)); 
        }
    
        public static boolean areConsecutive(int a, int b, int c) {
            int[] numbers = {a, b, c};
            Arrays.sort(numbers);
            
            return (numbers[1] == numbers[0] + 1) && (numbers[2] == numbers[1] + 1);
        }
    }
