    import java.util.Scanner;

    public class IdentityMatrixChecker {
        public static void main(String[] args) {
            Scanner scanner = new Scanner(System.in);
    
            System.out.print("Enter the size of the matrix (n for an n x n matrix): ");
            int n = scanner.nextInt();
    
            int[][] matrix = new int[n][n];
            
            System.out.println("Enter elements of the matrix:");
            for (int i = 0; i < n; i++) {
                for (int j = 0; j < n; j++) {
                    System.out.printf("Element [%d][%d]: ", i, j);
                    matrix[i][j] = scanner.nextInt();
                }
            }
    
            boolean isIdentity = true;
    
            for (int i = 0; i < n; i++) {
                for (int j = 0; j < n; j++) {
                    if ((i == j && matrix[i][j] != 1) || (i != j && matrix[i][j] != 0)) {
                        isIdentity = false;
                        break;
                    }
                }
                if (!isIdentity) {
                    break;
                }
            }
    
            if (isIdentity) {
                System.out.println("The given matrix is an identity matrix.");
            } else {
                System.out.println("The given matrix is NOT an identity matrix.");
            }
    
            scanner.close();
        }
    }
