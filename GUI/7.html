    import java.awt.*;

    public class CheckboxCustomLayout extends Frame {
        public CheckboxCustomLayout() {
            setLayout(null);
            Checkbox cb = new Checkbox("Check Me");
            cb.setBounds(50, 50, 100, 50);
            add(cb);
            setSize(200, 200);
            setVisible(true);
        }
        public static void main(String[] args) {
            new CheckboxCustomLayout();
        }
    }
