    import java.awt.*;

    public class LabelFlowLayout extends Frame {
        public LabelFlowLayout() {
            setLayout(new FlowLayout());
            add(new Label("This is a label"));
            setSize(200, 100);
            setVisible(true);
        }
        public static void main(String[] args) {
            new LabelFlowLayout();
        }
    }
