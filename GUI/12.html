    import java.awt.*;

    public class CheckboxGroupFlowLayout extends Frame {
        public CheckboxGroupFlowLayout() {
            setLayout(new FlowLayout());
            CheckboxGroup group = new CheckboxGroup();
            add(new Checkbox("Option 1", group, false));
            add(new Checkbox("Option 2", group, false));
            setSize(200, 100);
            setVisible(true);
        }
        public static void main(String[] args) {
            new CheckboxGroupFlowLayout();
        }
    }
