    import java.awt.*;
    import java.awt.event.*;
    
    public class ButtonCustomLayout extends Frame {
        public ButtonCustomLayout() {
            setLayout(null);
            Button button = new Button("Click Me");
            button.setBounds(50, 50, 100, 50);
            add(button);
            setSize(200, 200);
            setVisible(true);
        }
        public static void main(String[] args) {
            new ButtonCustomLayout();
        }
    }
