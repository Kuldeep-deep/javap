    import java.awt.*;

    public class ListFlowLayout extends Frame {
        public ListFlowLayout() {
            setLayout(new FlowLayout());
            List list = new List();
            list.add("Item 1");
            list.add("Item 2");
            list.add("Item 3");
            add(list);
            setSize(200, 150);
            setVisible(true);
        }
        public static void main(String[] args) {
            new ListFlowLayout();
        }
    }
