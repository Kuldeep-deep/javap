    import java.awt.*;

    public class TextAreaFlowLayout extends Frame {
        public TextAreaFlowLayout() {
            setLayout(new FlowLayout());

            add(textArea);
            setSize(250, 200);
            setVisible(true);
        }
        public static void main(String[] args) {
            new TextAreaFlowLayout();
        }
    }
