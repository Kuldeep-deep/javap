    import java.awt.*;

    public class TextAreaCustomLayout extends Frame {
        public TextAreaCustomLayout() {
            setLayout(null);
            TextArea textArea = new TextArea("Enter text here");
            textArea.setBounds(50, 50, 150, 100);
            add(textArea);
            setSize(250, 250);
            setVisible(true);
        }
        public static void main(String[] args) {
            new TextAreaCustomLayout();
        }
    }
