    import java.awt.*;

    public class ChoiceFlowLayout extends Frame {
        public ChoiceFlowLayout() {
            setLayout(new FlowLayout());
            Choice choice = new Choice();
            choice.add("Option 1");
            choice.add("Option 2");
            choice.add("Option 3");
            add(choice);
            setSize(200, 100);
            setVisible(true);
        }
        public static void main(String[] args) {
            new ChoiceFlowLayout();
        }
    }
