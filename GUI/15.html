    import java.awt.*;

    public class ListCustomLayout extends Frame {
        public ListCustomLayout() {
            setLayout(null);
            List list = new List();
            list.add("Item 1");
            list.add("Item 2");
            list.add("Item 3");
            list.setBounds(50, 50, 100, 60);
            add(list);
            setSize(200, 200);
            setVisible(true);
        }
        public static void main(String[] args) {
            new ListCustomLayout();
        }
    }
