    public class BooleanToString {
        public static void main(String[] args) {
            boolean boolValue = true;
    
            // Using String.valueOf()
            String str1 = String.valueOf(boolValue);
            System.out.println("String value using valueOf: " + str1);
    
            // Using Boolean.toString()
            String str2 = Boolean.toString(boolValue);
            System.out.println("String value using toString: " + str2);
        }
    }
    
    
OUTPUT

    String value using valueOf: true
    String value using toString: true
