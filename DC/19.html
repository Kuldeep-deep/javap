    public class CharToInt {
        public static void main(String[] args) {
            char charValue = 'A';
            int intValue = charValue;  // Implicit conversion to int (ASCII value of 'A')
            System.out.println("Integer value of '" + charValue + "': " + intValue);
        }
    }
    
    
OUTPUT

    Integer value of 'A': 65
