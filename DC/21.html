    public class StringToBoolean {
        public static void main(String[] args) {
            String str = "true";
            boolean boolValue = Boolean.parseBoolean(str);  
            System.out.println("Boolean value: " + boolValue);
            
            str = "FALSE";
            boolValue = Boolean.parseBoolean(str);  
            System.out.println("Boolean value: " + boolValue);
        }
    }
    
    
OUTPUT

    Boolean value: true
    Boolean value: false
